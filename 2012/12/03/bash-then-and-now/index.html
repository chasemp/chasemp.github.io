bash aliases

alias ips='ip brief'


#IP SHORTCUTS
ip () {
    case "$1" in
        'br')
            ip brief
            ;;
        'brief')
            sudo ip addr show | grep inet | grep -v 127 | awk '{print $7, $2}' | grep -v '::'
            ;;
        'alias')
            echo $"Usage: ip 
                         [alias|br|brief]"
            ;;
        *)
        sudo /bin/ip $@
    esac
}
